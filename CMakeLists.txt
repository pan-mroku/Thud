CMAKE_MINIMUM_REQUIRED(VERSION 2.8)

PROJECT(Thud)

SET(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=gnu++11 -g")

FIND_PACKAGE(wxWidgets COMPONENTS core base REQUIRED)
FIND_PACKAGE(osg REQUIRED)
FIND_PACKAGE(osgViewer REQUIRED)
FIND_PACKAGE(osgDB REQUIRED)

INCLUDE(${wxWidgets_USE_FILE})
INCLUDE(${OSG_INCLUDE_DIR})
INCLUDE(${OSGVIEWER_INCLUDE_DIR})
INCLUDE(${OSGDB_INCLUDE_DIR})

SET(HEADERS mainwindow.h gui.h app.hpp osgcontrol.hpp)
SET(SOURCES mainwindow.cpp gui.cpp app.cpp osgcontrol.cpp)

ADD_EXECUTABLE(${PROJECT_NAME}
  ${HEADERS}
  ${SOURCES}
  )

TARGET_LINK_LIBRARIES(${PROJECT_NAME} ${wxWidgets_LIBRARIES} ${OSGVIEWER_LIBRARIES} ${OSG_LIBRARIES})
